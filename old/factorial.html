Warning : Js doesnt seem to handle factorial over 170 -> it returns infinity (number is too large)


// Recursive
function rFact(num)
{
    if (num === 0)
      { return 1; }
    else
      { return num * rFact( num - 1 ); }
}

// Iterative
function sFact(num)
{
    var rval=1;
    for (var i = 2; i <= num; i++)
        rval = rval * i;
    return rval;
}
